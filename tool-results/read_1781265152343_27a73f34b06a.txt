   129→async function apiFetch<T>(url: string, options?: RequestInit): Promise<T> {
   130→  const res = await fetch(url, { headers: { 'Content-Type': 'application/json', ...options?.headers }, ...options })
   131→  const data = await res.json()
   132→  if (!res.ok) throw new Error(data.error || `API error ${res.status}`)
   133→  return data
   134→}

   553→function SearchTab() {
   554→  const [query, setQuery] = useState('')
   555→  const [results, setResults] = useState<SearchResultItem[]>([])
   556→  const [searching, setSearching] = useState(false)
   557→  const [ctxResult, setCtxResult] = useState<ContextResult | null>(null)

   559→  const [similarResults, setSimilarResults] = useState<SearchResultItem[]>([])
   560→
   561→  const doSearch = async () => {
   562→    if (!query.trim()) return
   563→    setSearching(true); setCtxResult(null)
   564→    try {
   565→      const data = await apiFetch<{ results: SearchResultItem[] }>('/api/knowledge/search', {
   566→        method: 'POST', body: JSON.stringify({ query, limit: 5 }),
   567→      })
   568→      setResults(data.results)
   569→    } catch { toast.error('Search failed') }
   570→    finally { setSearching(false) }
   571→  }
   572→
   573→  const doContext = async () => {
   574→    if (!query.trim()) return
   575→    setSearching(true)
   576→    try {
   577→      const data = await apiFetch<ContextResult>('/api/knowledge/context', {
   578→        method: 'POST', body: JSON.stringify({ query, maxDocuments: 5, maxTokenBudget: 5000 }),
   579→      })
   580→      setCtxResult(data)
   581→    } catch { toast.error('Context build failed') }
   582→    finally { setSearching(false) }
   583→  }
   584→
   585→  const doSimilar = async () => {
   586→    if (!similarId.trim()) return
   587→    setSearching(true)
   588→    try {
   589→      const data = await apiFetch<{ results: SearchResultItem[] }>(`/api/knowledge/similar?id=${similarId}&limit=5`)
   590→      setSimilarResults(data.results)
   591→    } catch { toast.error('Similar search failed') }
   592→    finally { setSearching(false) }
   593→  }
   594→
   595→  return (
   596→    <div className="space-y-4">
   597→      <div><h1 className="text-xl font-bold">Search & Retrieve</h1><p className="text-muted-foreground text-xs">Hybrid retrieval with score breakdown</p></div>
   598→
   599→      <Card>
   600→        <CardContent className="pt-4 space-y-3">
   601→          <div className="flex gap-2">
   602→            <Input placeholder="Search knowledge..." value={query} onChange={e => setQuery(e.target.value)} onKeyDown={e => e.key === 'Enter' && doSearch()} className="text-xs h-8" />
   603→            <Button size="sm" onClick={doSearch} disabled={searching}>{searching ? <Loader2 className="h-3 w-3 animate-spin" /> : <Search className="h-3 w-3 mr-1" />}Search</Button>
   604→            <Button size="sm" variant="outline" onClick={doContext} disabled={searching}><Brain className="h-3 w-3 mr-1" />Context</Button>
   605→          </div>
   606→        </CardContent>
   607→      </Card>
   608→
   609→      {results.length > 0 && (
   610→        <div className="space-y-2">
   611→          <h3 className="text-sm font-medium">Search Results</h3>
   612→          {results.map(r => (
   613→            <Card key={r.id} className="border-l-4" style={{ borderLeftColor: CAT_BAR[r.category] || '#6b7280' }}>
   614→              <CardContent className="py-2.5 px-3 space-y-1.5">
   615→                <div className="flex items-start justify-between gap-2">
   616→                  <div><div className="text-xs font-medium">{r.title}</div><div className="text-[10px] text-muted-foreground line-clamp-2">{r.description}</div></div>
   617→                  <div className="flex items-center gap-1 shrink-0"><CatBadge category={r.category} /><Badge variant="secondary" className="text-[10px]">{r.score.toFixed(3)}</Badge></div>
   618→                </div>
   619→                {/* Score breakdown */}

   674→function IngestTab() {

   703→      <div className="grid grid-cols-1 sm:grid-cols-2 gap-3">

   723→          <CardHeader className="pb-2"><CardTitle className="text-sm">Last Ingestion Result</CardTitle></CardHeader>
